def add_business_days(date, num)
  d = date
  num.times do |i|
    if d.friday?
      d += 3
    elsif d.saturday?
      d += 2
    else
      d += 1
    end
  end
  return d
end

    set_date(params) click to toggle source

    sets the date that can be used in the text field. The function allows incrementation of date, and ignore weekends /while incrementing

    # File lib/session.rb, line 85
def set_date(params)
  begin
    if @driver.find_element(:class, 'business-date').displayed?
      date_text = @driver.find_element(:class, 'business-date').text
      date = Date.parse(date_text)
    end
    begin
      if params[0] == '+'
        if params.include? "ignore-weekends"
          date = add_business_days(date, params[1].to_i)
        else
          date += params[1].to_i
        end
      elsif params[0] == '-'
        if params.include? "ignore-weekends"
          date = subtract_business_days(date, params[1].to_i)
        else
          date -= params[1].to_i
        end
      end
      @date = date.strftime('%d/%m/%y')
    rescue => ex
      $results.fail("date #{params.join(' ')}", ex)
    end
  rescue => ex
    $results.fail("date #{params.join(' ')}", ex)
  end
end

def subtract_business_days(date, num)
  d = date
  num.times do |_i|
    if d.monday?
      d -= 3
    elsif d.sunday?
      d -= 2
    else
      d -= 1
    end
  end
  return d
end

def table_wait
  begin
    $results.log_action('tableWait')
    data_processing = @driver.find_element(:id, 'data_processing')
    unless data_processing.displayed?
      $session.wait_until(data_processing.displayed?)
    end
    $session.wait_until(!data_processing.displayed?)
  rescue => ex
    $session.success = false
    $results.fail('tableWait', ex)
  end
end
